303  return Rect(0,0,0,0);

306  cut_rect.y = 0;
307  cut_rect.width = glayer->graph.width;

312  return cut_rect;

363  cut_rect.width -= (temp-(rect.x+rect.width));

464 LCUI_API int GraphLayer_GetGraph( LCUI_GraphLayer *ctnr,
465  LCUI_Graph *graph_buff,
466  LCUI_Rect rect )
467 {
468  static int i, total;
469  static uchar_t tmp_alpha, alpha;
470  static LCUI_Pos pos, glayer_pos;
471  static LCUI_GraphLayer *glayer;
472  static LCUI_Queue glayerQ;
473  static LCUI_Rect valid_area;
474  static LCUI_Graph tmp_graph;
475 
476  /* 检测这个区域是否有效 */
477  if (rect.x < 0 || rect.y < 0) {
478  return -1;
479  }
480  if (rect.x + rect.width > ctnr->graph.width
481  || rect.y + rect.height > ctnr->graph.height ) {
482  return -1;
483  }
484  if (rect.width <= 0 || rect.height <= 0) {
485  return -2;
486  }
487 
488  Graph_Init( &tmp_graph );
489  Queue_Init( &glayerQ, 0, NULL);
490  Queue_UsingPointer( &glayerQ );
491 
492  graph_buff->have_alpha = FALSE;
493  tmp_graph.have_alpha = TRUE;
494  Graph_Create( graph_buff, rect.width, rect.height );
495  GraphLayer_GetLayers( ctnr, rect, &glayerQ );
496  total = Queue_GetTotal( &glayerQ );
497  //_DEBUG_MSG( "total: %d\n", total );
498  if( total <= 0 ) {
499  Graph_Cut ( &ctnr->graph, rect, graph_buff );
500  Queue_Destroy( &glayerQ );
501  return 0;
502  }
503 
504  for(i=total-1; i>=0; --i) {
505  glayer = Queue_Get( &glayerQ, i );
506  valid_area = GraphLayer_GetValidRect( ctnr, glayer );
507  glayer_pos = GraphLayer_GetGlobalPos( ctnr, glayer );
508  valid_area.x += glayer_pos.x;
509  valid_area.y += glayer_pos.y;
510  switch( Graph_IsOpaque( &glayer->graph ) ) {
511  case -1:
512  Queue_DeletePointer( &glayerQ, i );
513  break;
514  case 0: break;
515  case 1:
516  if( LCUIRect_IncludeRect(valid_area, rect) ) {
517  goto skip_loop;
518  }
519  break;
520  default:break;
521  }
522  }
523 skip_loop:
524  total = Queue_GetTotal( &glayerQ );
525  //_DEBUG_MSG( "total: %d\n", total );
526  if(i <= 0){
527  Graph_Cut (&ctnr->graph, rect, graph_buff );
528  }
529  for(i=0; i<total; ++i) {
530  glayer = Queue_Get( &glayerQ, i );
531  //_DEBUG_MSG("%p = Queue_Get( %p, %d )\n", glayer, &glayerQ, i);
532  if( !glayer ) {
533  continue;
534  }
535  Graph_Lock( &glayer->graph );
536  pos = GraphLayer_GetGlobalPos( ctnr, glayer );
537  valid_area = GraphLayer_GetValidRect( ctnr, glayer );
538  Graph_Quote( &tmp_graph, &glayer->graph, valid_area );
539  //_DEBUG_MSG("valid area: %d,%d,%d,%d, pos: %d,%d, size: %d,%d\n",
540  // valid_area.x, valid_area.y, valid_area.width, valid_area.height,
541  // pos.x, pos.y, glayer->graph.width, glayer->graph.height
542  // );
543 
544  /* 获取相对坐标 */
545  pos.x = pos.x - rect.x + valid_area.x;
546  pos.y = pos.y - rect.y + valid_area.y;
547  //_DEBUG_MSG("mix pos: %d,%d\n", pos.x, pos.y);
548  /* 如果该图层没有继承父图层的透明度 */
549  if( !glayer->inherit_alpha ) {
550  Graph_Mix( graph_buff, &tmp_graph, pos );
551  } else {
552  /* 否则，计算该图层应有的透明度 */
553  alpha = GraphLayer_GetRealAlpha( glayer );
554  tmp_alpha = glayer->graph.alpha;
555  glayer->graph.alpha = alpha;
556  Graph_Mix( graph_buff, &tmp_graph, pos );
557  glayer->graph.alpha = tmp_alpha;
558  }
559  Graph_Unlock( &glayer->graph );
560  }
561  Queue_Destroy( &glayerQ );
562  return 0;
563 }

565 LCUI_API int GraphLayer_Front( LCUI_GraphLayer *glayer )
566 {

593  break;
594  }
595  }
596  }
597  /* 没有找到就退出 */
598  if( des_pos == -1 || src_pos == -1 ) {
599  return -2;
600  }
601  if( src_pos+1 == des_pos ) {
602  return 1;
603  }
604  /* 找到的话就移动位置 */
605  Queue_Move( child_list, des_pos, src_pos );
